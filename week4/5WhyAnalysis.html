                            color_changes AS (
                                SELECT

                                JOIN random_samples AS rs

        SELECT pixel_color, COUNT(*) AS occurrences

        WHERE x = {res[0]} AND y = {res[1]}
        GROUP BY pixel_color
        ORDER BY occurrences DESC
        LIMIT 5;
    """).fetchall()
